*&      Module  SET_ALV_0100  OUTPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
MODULE set_alv_0100 OUTPUT.
  IF go_con IS INITIAL.
    PERFORM create_alv CHANGING go_con go_alv.

    PERFORM set_light.
    PERFORM set_rejected_order_color.
    PERFORM set_layo USING 'X'
                         CHANGING gs_layo.
    PERFORM set_fcat USING 'X' CHANGING gt_fcat.
    PERFORM set_sort USING 'X' CHANGING gt_sort.

    PERFORM display_alv USING go_alv 'ZSPB1020' CHANGING gs_layo gt_data gt_fcat gt_sort.
  ELSE.
    PERFORM set_light.
    CALL METHOD go_alv->refresh_table_display
      EXCEPTIONS
        finished = 1
        OTHERS   = 2.
  ENDIF.
ENDMODULE.
