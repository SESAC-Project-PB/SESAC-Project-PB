MODULE user_command_0100 INPUT.
  DATA: lv_rcnt TYPE c LENGTH 2.
  CASE ok_code.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.

    WHEN 'SEARCH'.
      CLEAR  gt_list.
      PERFORM get_product_list.

    WHEN 'CREATE'.
      CALL SCREEN 200.
      PERFORM get_product_list.
      CLEAR zspb5003. " 생성 조건 초기화
      CLEAR gv_pronr.

    WHEN 'UPDATE'.
      CLEAR gv_subrc.
      PERFORM get_selected_row USING '수정' CHANGING lv_rcnt gv_subrc.

      IF gv_subrc <> 0.
        CLEAR gv_subrc.
        RETURN.
      ENDIF.
      PERFORM set_default_0300.
      CALL SCREEN 300.
      PERFORM get_product_list.
      CLEAR gv_pronr.

    WHEN 'DETAIL'.
      PERFORM get_selected_row  USING '조회' CHANGING lv_rcnt gv_subrc.

      IF gv_subrc <> 0.
        CLEAR gv_subrc.
        RETURN.
      ENDIF.

      CALL SCREEN 400.
      CLEAR gv_pronr.

  ENDCASE.

ENDMODULE.
