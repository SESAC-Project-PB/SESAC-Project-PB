*&---------------------------------------------------------------------*
*&      Module  SET_ALV_0100  OUTPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
MODULE set_alv_0100 OUTPUT.
  IF go_con IS INITIAL.
    PERFORM create_alv USING 'CON' go_con go_alv.
    CREATE OBJECT go_handler.
    SET HANDLER go_handler->handler_hotspot_click FOR go_alv ACTIVATION 'X'.

    PERFORM get_so_list CHANGING gv_subrc.
    PERFORM set_rcv_stock.
    PERFORM set_all_stock_status CHANGING gt_data.

    PERFORM set_sort USING 'X' CHANGING gt_sort.
    PERFORM set_light.
    PERFORM set_alv_layout USING 'X'
                         CHANGING gs_layo.
    PERFORM set_fcat USING 'X'.
    PERFORM display_alv USING go_alv 'ZSPB1006'
                        CHANGING gs_layo gt_data gt_fcat gt_sort.
  ELSE.
    PERFORM set_rcv_stock.
    PERFORM set_all_stock_status CHANGING gt_data.
    PERFORM set_light.
    PERFORM refresh_alv USING go_alv.
  ENDIF.

ENDMODULE.

MODULE set_alv_0101 OUTPUT.
  IF go_con_stock IS INITIAL.
    PERFORM create_alv USING 'CON_STOCK' go_con_stock go_alv_stock.
    PERFORM set_stock_status CHANGING gt_stock.
    PERFORM set_sort USING 'A' CHANGING gt_sort_stock.
    PERFORM set_light_item.
    PERFORM set_alv_layout USING 'A'
                         CHANGING gs_layo_stock.
    PERFORM set_fcat USING 'A'.
    PERFORM display_alv USING go_alv_stock 'ZSPB1025'
                        CHANGING gs_layo_stock gt_stock gt_fcat_stock gt_sort_stock.
  ELSE.
    PERFORM set_stock_status CHANGING gt_stock.
    PERFORM set_light_item.
    PERFORM refresh_alv USING go_alv_stock.
  ENDIF.
