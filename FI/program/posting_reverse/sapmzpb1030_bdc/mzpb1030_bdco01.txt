*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
MODULE pa_check OUTPUT.

  GET PARAMETER ID 'ZEXT_CALL' FIELD gv_external_call.
  GET PARAMETER ID 'ZEXT_PONO' FIELD gv_num.

  DATA: ls_data TYPE ztpbcustom.

  IF gv_external_call = 'X'.
    SELECT SINGLE *
      FROM ztpbcustom
      INTO CORRESPONDING FIELDS OF ls_data
      WHERE cusnr = gv_num.
    MOVE-CORRESPONDING ls_data TO zspb3002.
  ENDIF.

ENDMODULE.
