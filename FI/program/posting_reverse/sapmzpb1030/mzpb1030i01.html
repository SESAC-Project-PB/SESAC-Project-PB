  ENDCASE.

MODULE user_command_0100 INPUT.

  CASE ok_code.
    WHEN 'RAD1'.
      PERFORM get_list.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.
    WHEN 'ENTER'.
      CLEAR gv_fullname.
      PERFORM get_empname USING    zspb3001-pernr
                          CHANGING zspb3001-lname
                                   zspb3001-fname
                                   gv_fullname.

    WHEN 'SEARCH'.
      CLEAR gv_fullname.
      PERFORM get_list.
      PERFORM get_empname USING    zspb3001-pernr
                          CHANGING zspb3001-lname
                                   zspb3001-fname
                                   gv_fullname.

    WHEN 'CREATE'.
      zspb3002-waers = 'KRW'.
      zspb3002-pflag = ''.
      CLEAR: zspb3001, gv_fullname.
      PERFORM get_list.
      PERFORM alv_refresh.

      CALL SCREEN 200.
      CLEAR: zspb3001, zspb3002, gv_fullname.
      zspb3001-pflag = 'X'.
      PERFORM get_list.

    WHEN 'UPDATE'.

      PERFORM set_screen_0300.
      CLEAR: zspb3002, gv_fullname.
      zspb3001-pflag = 'X'.
      PERFORM get_list.

    WHEN 'DETAIL'.

      PERFORM set_screen_0400.
      zspb3001-pflag = 'X'.
      PERFORM get_list.

    WHEN OTHERS.
  ENDCASE.

ENDMODULE.
