*&---------------------------------------------------------------------*
*&      Form  SET_LIGHT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
*  -->  p1        text
*  <--  p2        text
*----------------------------------------------------------------------*
FORM set_light .

  LOOP AT gt_jitem_list INTO gs_jitem_list.
    " 미결 상태인 경우
    IF gs_jitem_list-cflag = ''.
      gs_jitem_list-light = '1'. " 빨간불(미결)

      " 반제 상태인 경우
    ELSEIF gs_jitem_list-cflag = 'X'.
      gs_jitem_list-light = '3'. " 초록불(반제)

    ENDIF.
    MODIFY gt_jitem_list FROM gs_jitem_list.
  ENDLOOP.

ENDFORM.
