MODULE set_alv_0100 OUTPUT.
  IF go_con_0100 IS INITIAL.
    PERFORM: init_alv      USING    go_con_0100
                                    go_alv_0100
                                    'CON_0100',

             req_event_for_alv,

             set_layout_h  USING    'A'
                                    'LIGHT'
                                    'X'
                                    '송장 리스트'(t02)
                           CHANGING gs_layo_h,

             set_fcat_h    USING    'ZSPB2012'
                           CHANGING gt_fcat_h,

             display_alv   USING    go_alv_0100
                                    ''
                                    gs_layo_h
                                    gt_fcat_h
                                    gt_list_h.
  ELSE.
    IF ok_code EQ 'SEARCH'.
      PERFORM set_fcat_h     USING    'ZSPB2012'
                             CHANGING gt_fcat_h.
    ENDIF.

    PERFORM refresh_alv        USING     go_alv_0100
                                         gs_layo_h
                                         gt_fcat_h.

  ENDIF.

ENDMODULE.
