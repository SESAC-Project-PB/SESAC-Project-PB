MODULE exit INPUT.
  CASE ok_code.
    WHEN 'EXIT'.
      LEAVE PROGRAM.

    WHEN 'CANC'.
      LEAVE TO SCREEN 0.

  ENDCASE.

ENDMODULE.
*&---------------------------------------------------------------------*
*&      Module  USER_COMMAND_0100  INPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
MODULE user_command_0100 INPUT.
  CASE ok_code.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.

    WHEN 'SEARCH'.
      " 송장 조회
      PERFORM search_data.

    WHEN 'INSERT'.
      " 송장 생성
      PERFORM insert_data.

    WHEN 'DETAIL'.
      " 송장 상세정보
      PERFORM detail_data.

    WHEN 'INFO'.
      CALL SCREEN 0400 STARTING AT 62 1 ENDING AT 81 3.

  ENDCASE.

ENDMODULE.

MODULE user_command_0200 INPUT.
  CASE ok_code.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.

    WHEN 'ITEM_ALL'.
      " 전체 아이템 등록
      PERFORM insert_item.

    WHEN 'ITEM_DE'.
      " 아이템 삭제
      PERFORM delete_item USING    '삭제'
                          CHANGING  gt_display_item
                                    gv_subrc.
    WHEN 'INSERT'.
      " 송장 생성
      PERFORM insert_inv.

  ENDCASE.

ENDMODULE.
