MODULE exit INPUT.
  CASE ok_code.
    WHEN 'EXIT'.
      LEAVE PROGRAM.

    WHEN 'CANC'.
      LEAVE TO SCREEN 0.

  ENDCASE.

ENDMODULE.

*&      Module  USER_COMMAND_0100  INPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
MODULE user_command_0100 INPUT.
  CASE ok_code.
    WHEN 'ITYPE'.
      PERFORM clear_condition.

    WHEN 'BACK'.
      LEAVE TO SCREEN 0.

    WHEN 'SEARCH'.
      " 송장 조회
      PERFORM search_data.

    WHEN 'APPROVE'.
      " 송장 승인
      PERFORM approve.

    WHEN 'CANCEL'.
      " 송장 취소
      PERFORM cancel.

    WHEN 'INFO'.
      CALL SCREEN 0200 STARTING AT 58 1 ENDING AT 79 1.

  ENDCASE.

ENDMODULE.
